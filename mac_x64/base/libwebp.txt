export MACOSX_DEPLOYMENT_TARGET=10.13

--disable-tiff --disable-gif \

--with-jpegincludedir=/Applications/EServer/Library/jpeg/include \
--with-jpeglibdir=/Applications/EServer/Library/jpeg/lib

make -j8
